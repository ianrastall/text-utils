    <script>

        'use strict';



        const categorySelect = document.getElementById('categorySelect');

        const codeFormatSelect = document.getElementById('codeFormatSelect');

        const iconsGrid = document.getElementById('iconsGrid');

        const iconCount = document.getElementById('iconCount');

        const outputSnippet = document.getElementById('outputSnippet');

        const statusMessage = document.getElementById('statusMessage');

        const toolStats = document.getElementById('toolStats');

        const selectedName = document.getElementById('selected-name');

        const selectedCodepoint = document.getElementById('selected-codepoint');

        const tagList = document.getElementById('tagList');



        let currentIcons = [];

        let selectedIcon = null;



        function setStatus(msg, type = 'success') {

            statusMessage.textContent = msg;

            statusMessage.className = type;

        }



        function setPlaceholder(text) {

            iconsGrid.innerHTML = `<div class="muted" style="grid-column:1/-1; text-align:center;">${text}</div>`;

        }



        function loadCategories() {

            const files = [

                'accessibility.txt','brand.txt','chart.txt','currency.txt','directional.txt','file-type.txt','form-control.txt','gender.txt','hand.txt','payment.txt','spinner.txt','text-editor.txt','transportation.txt','video-player.txt','web-application.txt'

            ];

            categorySelect.innerHTML = '<option value="">-- Select a Category --</option>';

            files.sort().forEach(file => {

                const name = file.replace('.txt','').replace(/-/g,' ');

                const option = document.createElement('option');

                option.value = file;

                option.textContent = name.charAt(0).toUpperCase() + name.slice(1);

                categorySelect.appendChild(option);

            });

            setStatus('Ready', 'success');

            toolStats.textContent = 'Select a category to start';

        }



        function loadIcons(file) {

            if (!file) {

                currentIcons = [];

                iconCount.textContent = '0';

                setPlaceholder('Select a category to view icons.');

                return;

            }

            setStatus('Loading icons.', 'warning');

            iconsGrid.innerHTML = '<div class="muted" style="grid-column:1/-1;text-align:center;">Loadingâ€¦</div>';

            fetch(`font-awesome-categories/${file}`)

                .then(res => { if (!res.ok) throw new Error('Failed to load icons'); return res.text(); })

                .then(text => {

                    currentIcons = text.split(/\r?\n/).filter(Boolean).map(line => {

                        const [name, codepoint] = line.split(/\s+/);

                        return { name, codepoint };

                    });

                    renderIcons();

                    setStatus('Icons loaded', 'success');

                    toolStats.textContent = `${currentIcons.length} icons loaded`;

                })

                .catch(err => {

                    console.error(err);

                    setStatus('Failed to load icons', 'error');

                    setPlaceholder('Unable to load icons.');

                });

        }



        function renderIcons() {

            iconsGrid.innerHTML = '';

            iconCount.textContent = currentIcons.length;

            if (!currentIcons.length) {

                setPlaceholder('No icons found.');

                return;

            }

            currentIcons.forEach(icon => {

                const btn = document.createElement('button');

                btn.type = 'button';

                btn.className = 'icon-item';

                btn.setAttribute('aria-label', icon.name);

                btn.innerHTML = `<i class="fa fa-${icon.name}" aria-hidden="true"></i>`;

                btn.addEventListener('click', () => selectIcon(icon, btn));

                iconsGrid.appendChild(btn);

            });

        }



        function selectIcon(icon, btn) {

            iconsGrid.querySelectorAll('.icon-item').forEach(el => el.classList.remove('active'));

            btn.classList.add('active');

            selectedIcon = icon;

            selectedName.textContent = icon.name;

            selectedCodepoint.textContent = icon.codepoint || '-';

            tagList.innerHTML = '';

            ['fa', `fa-${icon.name}`].forEach(tag => {

                const chip = document.createElement('span');

                chip.className = 'chip';

                chip.textContent = tag;

                tagList.appendChild(chip);

            });

            outputSnippet.textContent = buildSnippet(icon);

            setStatus(`Selected ${icon.name}`, 'success');

        }



        function buildSnippet(icon) {

            switch (codeFormatSelect.value) {

                case 'html':

                    return `<i class="fa fa-${icon.name}" aria-hidden="true"></i>`;

                case 'css_before':

                    return `.icon-${icon.name}::before { content: "\\${(icon.codepoint || '').replace('&#x','').replace(';','')}"; font-family: "FontAwesome"; }`;

                case 'react':

                    return `<i className="fa fa-${icon.name}" aria-hidden="true"></i>`;

                case 'js_dom':

                    return `const icon = document.createElement('i');\nicon.className = 'fa fa-${icon.name}';\nicon.setAttribute('aria-hidden','true');\nelement.appendChild(icon);`;

                case 'icon_name':

                    return icon.name;

                case 'codepoint':

                    return icon.codepoint || '';

                default:

                    return '';

            }

        }



        function copySnippet() {

            if (!outputSnippet.textContent) {

                setStatus('Nothing to copy', 'warning');

                return;

            }

            navigator.clipboard.writeText(outputSnippet.textContent).then(

                () => setStatus('Copied!', 'success'),

                () => setStatus('Copy failed', 'error')

            );

        }



        function clearSelection() {

            selectedIcon = null;

            selectedName.textContent = 'None';

            selectedCodepoint.textContent = '-';

            outputSnippet.textContent = '';

            tagList.innerHTML = '';

            iconsGrid.querySelectorAll('.icon-item').forEach(el => el.classList.remove('active'));

            setStatus('Cleared', 'success');

        }



        document.getElementById('backButton').addEventListener('click', () => window.location.href = 'index.html');

        document.getElementById('copyBtn').addEventListener('click', copySnippet);

        document.getElementById('clearBtn').addEventListener('click', clearSelection);



        categorySelect.addEventListener('change', (e) => {

            clearSelection();

            loadIcons(e.target.value);

        });



        codeFormatSelect.addEventListener('change', () => {

            if (selectedIcon) {

                outputSnippet.textContent = buildSnippet(selectedIcon);

            }

        });



        loadCategories();

    </script>
